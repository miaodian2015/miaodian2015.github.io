package com.example.administrator.jiaoxueshu;

import android.app.Activity;
import android.content.Intent;
import android.support.v7.app.ActionBarActivity;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.RadioButton;

public class MainActivity extends Activity {

private EditText et_nm;
private EditText et_pw;
private RadioButton rb_man;
private RadioButton rb_woman;
private Button bt;

@Override
protected void onCreate(Bundle savedInstanceState) {
    super.onCreate(savedInstanceState);
    setContentView(R.layout.activity_main);
     et_nm = (EditText) findViewById(R.id.et_nm);
    et_pw= (EditText) findViewById(R.id.et_pw);
    rb_man= (RadioButton) findViewById(R.id.rb_man);
    rb_woman= (RadioButton) findViewById(R.id.rb_woman);
     bt = (Button) findViewById(R.id.bt);

    bt.setOnClickListener(new View.OnClickListener() {
        @Override
        public void onClick(View v) {
            passDate();
        }


    });

}

public void passDate() {
    //创建Intent对象，启动Activity2
    Intent intent=new Intent(this,Activity2.class);
    //将数据存入Intent对象
    intent.putExtra("name",et_nm.getText().toString().trim());
    intent.putExtra("password",et_pw.getText().toString().trim());
    String str="";
    if (rb_man.isChecked()){
        str="男";
    }else if (rb_woman.isChecked()){
        str="女";
    }
    intent.putExtra("sex",str);
    startActivity(intent);
}

}
